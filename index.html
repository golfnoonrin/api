📦 Request Format
{
  "token": "test_token_123",
  "sheetName": "Sheet1",
  "id": "optional_id",
  "options": {
    "limit": 10,
    "offset": 0,
    "filters": {
      "column1": "value1",
      "column2": "value2"
    }
  }
}

{
  "success": true,
  "data": [...],
  "total": 100,
  "limit": 10,
  "offset": 0
}

import requests
import json

<?php
$url = "https://script.google.com/macros/s/AKfycbzDv31Nv4ikHzovaYl2As5VEe7qpArBoPHap0ZSdGmfhKByvCw-b3p89-YEiARdKApf/exec";
$data = [
    "token" => "test_token_123",
    "sheetName" => "Sheet1",
    "options" => ["limit" => 10]
];

$options = [
    "http" => [
        "header" => "Content-Type: application/json",
        "method" => "POST",
        "content" => json_encode($data)
    ]
];

$context = stream_context_create($options);
$result = file_get_contents($url, false, $context);
echo $result;
?>
